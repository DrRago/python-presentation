Klassen


def typename(x):
  return f'{x.__class__.__name__}'

class Object:
  def __str__(self):
    return typename(self)

class Wrapper(Object):
  def __init__(self, o, *args, **kwargs):
    super().__init__(*args, **kwargs)
    self.o = o

o = Object()
w = Wrapper(3)

print(o)
print(w)
print(w.o - 3, end='\n')

Object.__str__ = lambda self: 'lul'

print(o)
print(w, end='\n')

Wrapper.__str__ = typename

print(o)
print(w)

Funktionsparameter


def print_keys(*keys, **d):
    for key in keys:
        value = d.get(key, "key doesn't exist")
        print(key, ':', value)

test = {
    '1': 1,
    '2': 'cool',
    '5': 3,
    '6': 2,
}

print_keys('1', '4', '5', '2', **test)
print_keys(*map(str, range(1, 6)), **test)

Generatoren


def fizz_buzz(x):
  rules = {
      2: 'fizz',
      5: 'buzz'
  }

  items = [rules.get(key, x) for key in rules.keys() if not x % key]

  if items:
    yield from items
  else:
    yield x

for i in (fizz_buzz(x + 1) for x in range(0, 20)):
  print(*i)

Anwendungsbeispiel


def factorisation(n, fac=2):
    if (n < fac * fac):
        l.append(n)
        return
    if (n % fac == 0):
        l.append(fac)
        factorisation(n // fac, fac)
    else:
        factorisation(n, fac + 1)


l = []
factorisation(100)
print(l)
